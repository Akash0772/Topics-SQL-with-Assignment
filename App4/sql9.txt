CREATE TABLE EMPLOYEE(ID INT UNIQUE, 
		      FIRST_NAME VARCHAR(90), 
		      LAST_NAME VARCHAR(90), 
		      DATE_BIRTH TIMESTAMP,
		      GENDER_ID INT,
		      LATEST_EDUCATION_ID INT,
		      JOINING_DATE DATE,
		      PROJECT_ID INT,
		      USERNAME VARCHAR(90),
		      PASSWORD VARCHAR(90),

select * from employee_skill;


select * from employee, address where employee.id = address.employee_id;

select * from employee, address, gender where 
employee.id = address.employee_id and 
employee.gender_id = gender.id;

select * from employee, address, gender, education where 
employee.id = address.employee_id and 
employee.gender_id = gender.id and 
employee.latest_education_id = education.id;

select * from employee, address, gender, education, project where 
employee.id = address.employee_id and 
employee.gender_id = gender.id and 
employee.latest_education_id = education.id and 
employee.project_id = project.id;

select * from employee, address, gender, education, project, employee_skill, skill where 
employee.id = address.employee_id and 
employee.gender_id = gender.id and 
employee.latest_education_id = education.id and 
employee.project_id = project.id and
employee.id = employee_skill.employee_id and
employee_skill.skill_id = skill.id;


SELECT E.ID, E.FIRST_NAME, E.LAST_NAME, E.DATE_BIRTH, E.JOINING_DATE, 
E.GENDER_ID, E.LATEST_EDUCATION_ID, E.PROJECT_ID, 
A.HOUSE_NO, A.STREET_NAME, A.CITY, A.STATE, 
G.NAME GENDER_NAME, P.NAME PROJECT_NAME, LE.NAME EDUCATION_NAME, S.ID SKILL_ID, 
S.NAME SKILL_NAME 
FROM EMPLOYEE E, ADDRESS A, GENDER G, PROJECT P,  
EDUCATION LE, EMPLOYEE_SKILL ES, SKILL S 
WHERE E.ID = A.EMPLOYEE_ID AND 
E.GENDER_ID = G.ID AND 
E.LATEST_EDUCATION_ID = LE.ID AND 
E.PROJECT_ID = P.ID AND  	     
E.ID = ES.EMPLOYEE_ID AND 
ES.SKILL_ID = S.ID AND 
E.ID = 22
